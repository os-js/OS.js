Tutorial: Create your own application

Create from template

Simply run:

./obt create-package MyName

This will create some source files for you with usable code. You'll have a basic running application with a Window.

Update manifest

The manifest holds information about your package, like name, description, icon and used resources. Update package.json to fit your application. 

{
  "className":  "ApplicationMyName",               // ClassName used in namespaces
  "enabled":    true,                              // If 'false' application will not be loaded or launchable (Optional)
  "description":"Application Long Description",    // Application description (optional)
  "name":       "Application Name",                // Application name
  "category":   "unknown",                         // Application category (optional)
  "mime":       null,                              // Or an array of regext match MIME strings (Optional)
  "icon":       "apps/file-manager.png",           // Application icon (use ./my-icon.png for icon inside app folder)
  "singular":   false,                             // Allow only one instance ? (Optional)
  "splash":     false,                             // Show a splash screen when loading application (Optional)
  "compability":[],                                // Array of features to check for (Optional, uses Utils.getCompability())
  "preload":    [                                  // A list of files to preload in order to run app
    {"src": "main.js", "type": "javascript"},
    {"src": "main.css", "type": "stylesheet"},
    {"src": "foo.js", "type": "javascript", combine: false} // Do not combine this script in 'dist'
  ]
}

// For translations use
{
  "names" : {
    "no_NO" : "Norwegian name"
  }
  "descriptions" : {
    "no_NO" : "Norwegian description"
  }
}

Now run this command to add it to global manifest (make it visible in OS.js)

make manifest

And, if you're using dist (prodiction environment)

make packages

Add a GUI Element

Example:

ApplicationMyNameWindow.prototype.init = function() {
  var self = this;
  var root = Window.prototype.init.apply(this, arguments);

  var listView = this._addGUIElement(new OSjs.GUI.ListView('MyListView'), root);
  listView.setColumns([
    {key: 'col1',    title: 'Foo', domProperties: {width: "50"}},
    {key: 'col2',   title: 'Bar'},
    {key: 'col3',  title: 'Baz', domProperties: {width: "100"}}
  ]);
  listView.render();

  return root;
};

To retrieve this element later, just use:

var gel = this._getGUIElement('MyListView');

Handling launch/restore arguments

Let's say you open your application with:

OSjs.API.launch('ApplicationMyName', {path: '/some/file.png', mime: 'image/png'});

Use this to handle it correctly:

ApplicationMyName.prototype.init = function(core, session) {
  Application.prototype.init.apply(this, arguments);

  // Get an argument, ex. the filename and mime
  var file = this._getArgument('file');
  if ( file && file.path ) {
    this.openFile(file);
  }
};

ApplicationMyName.prototype.openFile = function(file) {
  if ( file && !file.mime.match(/^image/) ) {
    throw "Unsupported MIME";
  }
  this.currentFile = fname;

  // Setting values
  this._setArgument('file', file);
};

Adding Drag-and-Drop support

Example:

var ApplicationMyNameWindow = function(app, opts) {
  Window.apply(this, ['ApplicationMyNameWindow', opts, app]);
  this._icon = "apps/accessories-text-editor.png";
  this._title = "DnD Support";
  this._properties.allow_drop = true;
};


ApplicationMyNameWindow.prototype._onDndEvent = function(ev, type, item, args) {
  Window.prototype._onDndEvent.apply(this, arguments);
  if ( type === 'itemDrop' && item ) {
    var data = item.data;
    if ( data && data.type === 'file' && data.mime ) {
      this.openFile(data.path, data.mime);
    }
  }
};

More advanced examples

You can extend the classes DefaultApplication and DefaultApplicationWindow for super-simple handling of opening/saving and creating files. It will also handle dnd and launch/restore arguments for you. Almost all internal OS.js applications use these.
